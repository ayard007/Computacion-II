🔧 Funciones Esenciales del Inventario

Vamos a construir las 4 operaciones básicas (CRUD) como funciones individuales. Las usaremos en la Fase 2.

Python · Celda 8
# ── Base de datos en memoria (lista de diccionarios) ──
inventario = []

def agregar_refaccion(codigo, nombre, categoria, cantidad, precio, minimo=5):
    """Agrega una nueva refacción al inventario."""
    nueva = {
        "codigo"   : codigo,
        "nombre"   : nombre,
        "categoria": categoria,
        "cantidad" : cantidad,
        "precio"   : precio,
        "minimo"   : minimo
    }
    inventario.append(nueva)
    print(f"✅ Agregado: {nombre} ({codigo})")

def buscar_refaccion(termino):
    """Busca por código o nombre (insensible a mayúsculas)."""
    termino = termino.lower()
    resultados = [
        r for r in inventario
        if termino in r["codigo"].lower()
        or termino in r["nombre"].lower()
    ]
    return resultados

def actualizar_cantidad(codigo, nueva_cantidad):
    """Actualiza la cantidad de una refacción por código."""
    for r in inventario:
        if r["codigo"] == codigo:
            r["cantidad"] = nueva_cantidad
            print(f"✅ {r['nombre']}: cantidad actualizada a {nueva_cantidad}")
            return
    print(f"❌ Código {codigo} no encontrado")

def eliminar_refaccion(codigo):
    """Elimina una refacción del inventario."""
    global inventario
    original = len(inventario)
    inventario = [r for r in inventario if r["codigo"] != codigo]
    if len(inventario) < original:
        print(f"🗑️ Refacción {codigo} eliminada.")
    else:
        print(f"❌ No se encontró {codigo}")

# ── Probar las funciones ──
agregar_refaccion("REF-001", "SSD 500GB",    "Almacenamiento", 15, 850.0)
agregar_refaccion("REF-002", "RAM 8GB DDR4", "Memoria",        3,  420.0)
agregar_refaccion("REF-003", "Fuente 600W",  "Energía",         8,  650.0)

print("\n── Búsqueda 'ram' ──")
for r in buscar_refaccion("ram"):
    print(f"  {r['codigo']}: {r['nombre']}")

actualizar_cantidad("REF-002", 10)
¿Por qué usar funciones?

Las funciones hacen tu código modular — puedes reutilizar agregar_refaccion() en cualquier parte del programa sin reescribirla. En la Fase 2 las conectaremos directamente al Excel.

def cargar_inventario():
    """Carga el inventario desde Excel. Si no existe, crea uno vacío."""
    if os.path.exists(ARCHIVO):
        df = pd.read_excel(ARCHIVO, sheet_name=HOJA)
        print(f"✅ Inventario cargado: {len(df)} registros")
        return df

                    "cantidad", "precio", "minimo", "fecha_ingreso"]
        df = pd.DataFrame(columns=columnas)
        print("📄 Creando nuevo inventario...")
        return df

def agregar(df, codigo, nombre, categoria, cantidad, precio, minimo=5):
    """CREATE — agrega una refacción nueva."""
    if codigo in df["codigo"].values:
        print(f"❌ El código {codigo} ya existe.")
        return df
    nueva_fila = {
        "codigo"       : codigo,
        "nombre"       : nombre,
        "categoria"    : categoria,
        "cantidad"     : cantidad,
        "precio"       : precio,
        "minimo"       : minimo,
        "fecha_ingreso": datetime.today().strftime("%Y-%m-%d")
    }
    df = pd.concat([df, pd.DataFrame([nueva_fila])], ignore_index=True)
    guardar_inventario(df)
    print(f"✅ Agregado: {nombre}")
    return df

def ver_inventario(df):
    """READ — muestra todo el inventario con estado de stock."""
    print("\n" + "="*60)
    print("  INVENTARIO — TALLER TECH SERVICES")
    print("="*60)
    for _, row in df.iterrows():
        estado = "⚠ BAJO" if row["cantidad"] <= row["minimo"] else "  OK "
        print(f"[{estado}] {row['codigo']} | {row['nombre']:22} | "
              f"Cant: {int(row['cantidad']):>3} | Precio: ${row['precio']:>8,.2f}")
    print("-"*60)
    print(f"Total artículos: {len(df)} | Valor total: ${(df['cantidad']*df['precio']).sum():,.2f}\n")

def actualizar(df, codigo, campo, nuevo_valor):
    """UPDATE — modifica un campo de una refacción."""
    idx = df[df["codigo"] == codigo].index
    if idx.empty:
        print(f"❌ Código {codigo} no encontrado")
        return df
    df.loc[idx, campo] = nuevo_valor
    guardar_inventario(df)
    print(f"✅ {codigo} → {campo} = {nuevo_valor}")
    return df

def eliminar(df, codigo):
    """DELETE — elimina una refacción por código."""
    if codigo not in df["codigo"].values:
        print(f"❌ Código {codigo} no encontrado")
        return df
    nombre = df[df["codigo"]==codigo]["nombre"].values[0]
    df = df[df["codigo"] != codigo].reset_index(drop=True)
    guardar_inventario(df)
    print(f"🗑️ Eliminado: {nombre}")
    return df

def alertas_stock(df):
    """Muestra los artículos con stock igual o menor al mínimo."""
    bajos = df[df["cantidad"] <= df["minimo"]]
    sin_stock = df[df["cantidad"] == 0]

    print("\n🔔 === ALERTAS DE INVENTARIO ===")

    if not sin_stock.empty:
        print("\n🔴 SIN STOCK:")
        for _, r in sin_stock.iterrows():
            print(f"   • {r['codigo']}: {r['nombre']}")

    if not bajos.empty:
        print("\n🟡 STOCK BAJO (reponer pronto):")
        for _, r in bajos.iterrows():

def reporte_general(df):
    """Genera un reporte completo del inventario."""
    print("\n" + "="*55)
    print("  REPORTE GENERAL — TALLER TECH SERVICES")
    print(f"  Fecha: {datetime.today().strftime('%d/%m/%Y %H:%M')}")
    print("="*55)

    # KPIs principales
    total_articulos  = len(df)
    valor_inventario = (df["cantidad"] * df["precio"]).sum()
    art_bajo_stock   = len(df[df["cantidad"] <= df["minimo"]])
    precio_promedio  = df["precio"].mean()

    print(f"\n📦 Total artículos   : {total_articulos}")
    print(f"💰 Valor inventario  : ${valor_inventario:,.2f}")
    print(f"⚠  Bajo stock        : {art_bajo_stock} artículo(s)")
    print(f"📈 Precio promedio   : ${precio_promedio:,.2f}")

    # Reporte por categoría
    print("\n── Por Categoría ──")
    por_cat = df.groupby("categoria").agg(
        articulos   = ("nombre",   "count"),
        stock_total = ("cantidad", "sum"),
        valor_total = ("precio",   lambda x: (x * df.loc[x.index, "cantidad"]).sum())
    ).reset_index()

    for _, r in por_cat.iterrows():
        print(f"  {r['categoria']:18} | {r['articulos']:>2} arts | "
              f"Stock: {int(r['stock_total']):>4} | Valor: ${r['valor_total']:>10,.2f}")

    # Top 3 artículos más valiosos
    print("\n── Top 3 Mayor Valor en Almacén ──")
    df_val = df.copy()
    df_val["valor"] = df_val["cantidad"] * df_val["precio"]
    top3 = df_val.nlargest(3, "valor")
    for i, (_, r) in enumerate(top3.iterrows(), 1):
        print(f"  {i}. {r['nombre']:22} → ${r['valor']:>10,.2f}")

def exportar_reporte_excel(df):
    """Guarda un reporte detallado en una nueva hoja del Excel."""
    df_rep = df.copy()
    df_rep["valor_total"] = df_rep["cantidad"] * df_rep["precio"]
    df_rep["estado"] = df_rep.apply(
        lambda r: "SIN STOCK" if r["cantidad"]==0
                  else ("STOCK BAJO" if r["cantidad"] <= r["minimo"] else "OK"), axis=1)

    with pd.ExcelWriter(ARCHIVO, engine="openpyxl", mode="a",
                        if_sheet_exists="replace") as writer:
        df_rep.to_excel(writer, sheet_name="Reporte", index=False)

    print(f"📊 Reporte exportado a hoja 'Reporte' en {ARCHIVO}")
